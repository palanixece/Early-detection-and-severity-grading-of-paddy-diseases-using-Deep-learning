        "# ============================================================\n",
        "# CELL 6: SEVERITY CALCULATION ENGINE\n",
        "# ============================================================\n",
        "\n",
        "print(\"=\" * 60)\n",
        "print(\"SEVERITY CALCULATION ENGINE\")\n",
        "print(\"=\" * 60)\n",
        "\n",
        "def calculate_severity(image_np, method='hybrid'):\n",
        "    \"\"\"\n",
        "    Calculate disease severity from a rice leaf image.\n",
        "\n",
        "    Methods:\n",
        "    - 'otsu': Otsu's thresholding (same as the paper)\n",
        "    - 'hsv': HSV color-based segmentation\n",
        "    - 'hybrid': Combination of both (recommended)\n",
        "\n",
        "    Returns:\n",
        "    - severity_label: 0 (Mild), 1 (Moderate), 2 (Severe)\n",
        "    - severity_ratio: percentage of infected area\n",
        "    \"\"\"\n",
        "    if image_np.shape[-1] == 3:\n",
        "        img_bgr = cv2.cvtColor(image_np, cv2.COLOR_RGB2BGR)\n",
        "    else:\n",
        "        img_bgr = image_np.copy()\n",
        "\n",
        "    if method == 'otsu':\n",
        "        return _otsu_severity(img_bgr)\n",
        "    elif method == 'hsv':\n",
        "        return _hsv_severity(img_bgr)\n",
        "    elif method == 'hybrid':\n",
        "        return _hybrid_severity(img_bgr)\n",
        "\n",
        "\n",
        "def _otsu_severity(img_bgr):\n",
        "    \"\"\"Otsu's thresholding method (as described in the paper)\"\"\"\n",
        "    gray = cv2.cvtColor(img_bgr, cv2.COLOR_BGR2GRAY)\n",
        "    blurred = cv2.GaussianBlur(gray, (5, 5), 0)\n",
        "    _, binary_mask = cv2.threshold(\n",
        "        blurred, 0, 255, cv2.THRESH_BINARY_INV + cv2.THRESH_OTSU\n",
        "    )\n",
        "\n",
        "    kernel = np.ones((3, 3), np.uint8)\n",
        "    binary_mask = cv2.morphologyEx(binary_mask, cv2.MORPH_CLOSE, kernel)\n",
        "    binary_mask = cv2.morphologyEx(binary_mask, cv2.MORPH_OPEN, kernel)\n",
        "\n",
        "    total_pixels = binary_mask.size\n",
        "    infected_pixels = np.count_nonzero(binary_mask)\n",
        "\n",
        "    if total_pixels == 0:\n",
        "        return 0, 0.0\n",
        "\n",
        "    severity_ratio = (infected_pixels / total_pixels) * 100\n",
        "    return _ratio_to_label(severity_ratio), severity_ratio\n",
        "\n",
        "\n",
        "def _hsv_severity(img_bgr):\n",
        "    \"\"\"HSV color-based segmentation method\"\"\"\n",
        "    hsv = cv2.cvtColor(img_bgr, cv2.COLOR_BGR2HSV)\n",
        "\n",
        "    # Healthy leaf (green)\n",
        "    lower_green = np.array([20, 25, 25])\n",
        "    upper_green = np.array([95, 255, 255])\n",
        "    healthy_mask = cv2.inRange(hsv, lower_green, upper_green)\n",
        "\n",
        "    # Brown/dark spots\n",
        "    lower_brown = np.array([0, 20, 20])\n",
        "    upper_brown = np.array([20, 255, 180])\n",
        "    brown_mask = cv2.inRange(hsv, lower_brown, upper_brown)\n",
        "\n",
        "    # Yellow spots\n",
        "    lower_yellow = np.array([15, 40, 100])\n",
        "    upper_yellow = np.array([35, 255, 255])\n",
        "    yellow_mask = cv2.inRange(hsv, lower_yellow, upper_yellow)\n",
        "\n",
        "    # Dark lesions\n",
        "    lower_dark = np.array([0, 0, 0])\n",
        "    upper_dark = np.array([180, 255, 50])\n",
        "    dark_mask = cv2.inRange(hsv, lower_dark, upper_dark)\n",
        "\n",
        "    # Combine disease masks\n",
        "    disease_mask = cv2.bitwise_or(brown_mask, yellow_mask)\n",
        "    disease_mask = cv2.bitwise_or(disease_mask, dark_mask)\n",
        "\n",
        "    # Clean up\n",
        "    kernel = np.ones((5, 5), np.uint8)\n",
        "    healthy_mask = cv2.morphologyEx(healthy_mask, cv2.MORPH_CLOSE, kernel)\n",
        "    disease_mask = cv2.morphologyEx(disease_mask, cv2.MORPH_CLOSE, kernel)\n",
        "    disease_mask = cv2.morphologyEx(disease_mask, cv2.MORPH_OPEN, kernel)\n",
        "\n",
        "    total_leaf = np.count_nonzero(healthy_mask) + np.count_nonzero(disease_mask)\n",
        "    infected = np.count_nonzero(disease_mask)\n",
        "\n",
        "    if total_leaf == 0:\n",
        "        return 0, 0.0\n",

        "# ============================================================\n",
        "# CELL 26: IoU (Intersection over Union) ANALYSIS\n",
        "# ============================================================\n",
        "\n",
        "print(\"=\" * 60)\n",
        "print(\"IoU (INTERSECTION OVER UNION) ANALYSIS\")\n",
        "print(\"=\" * 60)\n",
        "\n",
        "gc.collect()\n",
        "\n",
        "def calculate_iou_per_class(y_true, y_pred, num_cls):\n",
        "    \"\"\"Calculate IoU for each class from classification predictions\"\"\"\n",
        "    iou_scores = {}\n",
        "    for cls in range(num_cls):\n",
        "        true_mask = (y_true == cls)\n",
        "        pred_mask = (y_pred == cls)\n",
        "\n",
        "        intersection = np.logical_and(true_mask, pred_mask).sum()\n",
        "        union = np.logical_or(true_mask, pred_mask).sum()\n",
        "\n",
        "        if union == 0:\n",
        "            iou_scores[cls] = 1.0\n",
        "        else:\n",
        "            iou_scores[cls] = intersection / union\n",
        "\n",
        "    return iou_scores\n",
        "\n",
        "\n",
        "def calculate_segmentation_iou(image_np):\n",
        "    \"\"\"Calculate IoU between Otsu and HSV segmentation methods\"\"\"\n",
        "    img_bgr = cv2.cvtColor(image_np, cv2.COLOR_RGB2BGR)\n",
        "\n",
        "    # Otsu segmentation\n",
        "    gray = cv2.cvtColor(img_bgr, cv2.COLOR_BGR2GRAY)\n",
        "    blurred = cv2.GaussianBlur(gray, (5, 5), 0)\n",
        "    _, otsu_mask = cv2.threshold(blurred, 0, 255,\n",
        "                                  cv2.THRESH_BINARY_INV + cv2.THRESH_OTSU)\n",
        "\n",
